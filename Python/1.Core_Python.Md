						    specialChar = 0

							if ( (ch >= 'a' and ch <= 'z') or 
							     (ch >= 'A' and ch <= 'Z') ):  

							    # To handle upper case letters 
							    ch = ch.lower() 

							    if (ch == 'a' or ch == 'e' or ch == 'i'

								vowels += 1
							    else: 
								consonant += 1

							elif (ch >= '0' and ch <= '9'): 
							    digit += 1
							else: 
							    specialChar += 1

						    print("Vowels:", vowels) 
						    print("Consonant:", consonant)  
						    print("Digit:", digit)
